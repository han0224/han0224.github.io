function solution(n) {
    const arr = new Array(n).fill(0).map((v, i)=>new Array(i+1).fill(0));
    let count = 1;
    let status = 'down'; // down, up, right
    let i = 0;
    let j = 0;
    const endNum = arr.reduce((acc, cur) => cur.length + acc, 0);
    while(count <= endNum){
        arr[i][j] = count++;
        if(status==='down'){
            i += 1;
            if(i === n || arr[i][j] > 0){
                i -= 1;
                j += 1;
                status = 'right'
            }
            continue;
        }
        if(status === 'right'){
            j += 1;
            if(arr[i][j] === undefined || arr[i][j] > 0){
                i -= 1;
                j -= 2;
                status = 'up'
            }
            continue;
        }   
        if(status === 'up'){
            i -= 1;
            j -= 1;
            if(i < 0 || j < 0 || !arr[i][j] === undefined || arr[i][j] > 0){
                i += 2;
                j += 1;
                status = 'down';
            }
            continue;
        }
    }
    return arr.reduce((acc, cur) => [...acc, ...cur], []);
}
